Peraturan Permainan Angka BOM

  1. Komputer memilih 1 angka BOM acak antara 1 sampai 100.
  2. Pemain menebak angka dalam kisaran saat ini. Jika menebak angka BOM, pemain kalah.
  3. Jika tebakan > angka BOM, kisaran selanjutnya menjadi [terendah .. tebakan-1].
  4. Jika tebakan < angka BOM, kisaran selanjutnya menjadi [tebakan+1 .. tertinggi].
  5. Angka yang sudah ditebak tidak lagi termasuk ke kisaran selanjutnya.
  6. Jika tersisa 1 angka (angka BOM), pemain menang karena berhasil menghindari semua angka non-BOM.
  7. Jika tersisa 2 angka, dan pemain berhasil menebak 1 angka yang bukan BOM, pemain menang.
  8. Pemain hanya punya 20 detik untuk menyelesaikan permainan sebelum BOM meledak!

from InquirerPy import inquirer
from pyfiglet import figlet_format
from rich.console import Console
import os
import random
import secrets
import string
import time

MERAH = "\033[31m"
KUNING = "\033[33m"
HIJAU = "\033[32m"
BIRU = "\033[34m"
WARNING = "\033[1;31m"
RESET = "\033[0m"

console = Console()

def clear():
    os.system('cls' if os.name == 'nt' else 'clear')

def tampilkan_judul():
    judul = figlet_format("Angka BOM", font="slant")
    print(MERAH + judul + RESET)

def show_rules():

    console.print("4. Jika tebakan < angka BOM, kisaran selanjutnya menjadi [tebakan+1 .. tertinggi].")

    console.print("6. Jika tersisa 1 angka (angka BOM), pemain menang karena berhasil menghindari semua angka non-BOM.")
    console.print("7. Jika tersisa 2 angka, dan pemain berhasil menebak 1 angka yang bukan BOM, pemain menang.")
    console.print("8. Pemain hanya punya 20 detik untuk menyelesaikan permainan sebelum BOM meledak!\n")

def dramatic_loss(reason):
    clear()
    tampilkan_judul()
    msg = figlet_format("KALAH!", font="big")
    console.print(MERAH + msg + RESET)
    console.print(f":boom: {reason}\n")
    inquirer.text(message="Tekan Enter untuk kembali ke menu...").execute()

def dramatic_win(secret_code):
    clear()
    tampilkan_judul()
    msg = figlet_format("MENANG!", font="big")
    console.print(HIJAU + msg + RESET)
    console.print("Selamat — Anda berhasil menghindari angka BOM dan memenangkan permainan!\n")
    console.print(f"Kode akses Secret Section Anda adalah: [bold]{secret_code}[/bold]\n")
    inquirer.text(message="Tekan Enter untuk kembali ke menu...").execute()

def generate_secret_code(length: int = 8) -> str:
    alphabet = string.ascii_uppercase + string.digits
    return ''.join(secrets.choice(alphabet) for _ in range(length))

def ask_number_in_range(low: int, high: int, guessed: list) -> int:
    prompt = f"Tebak sebuah angka antara {low} sampai {high}: "

    def _validate(result):
        try:
            val = int(result)
        except Exception:
            return "Masukkan harus berupa angka bulat."
        if val < low or val > high:
            return f"Masukkan harus berada dalam kisaran {low} sampai {high}."
        if val in guessed:
            return f"Angka {val} sudah ditebak sebelumnya. Coba angka lain."
        return True

    answer = inquirer.text(message=prompt, validate=_validate).execute()
    return int(answer)

def countdown(seconds_left):
    for i in range(seconds_left, 0, -1):
        clear()
        tampilkan_judul()
        console.print(f"[bold red]BOM akan meledak dalam {i} detik![/bold red]")
        time.sleep(1)

def play_game():
    clear()
    tampilkan_judul()

    low, high = 1, 100
    bomb = random.randint(low, high)
    guessed = []
    secret_code = None

    console.print("Permainan dimulai! Anda memiliki [bold red]20 detik[/bold red] untuk menghindari angka BOM!\n")
    start_time = time.time()

    countdown_started = False

    while True:
        elapsed = time.time() - start_time
        remaining_time = 20 - elapsed

        if remaining_time <= 0:
            dramatic_loss("Waktu habis! BOM meledak sebelum Anda sempat menebak semua angka.")
            break

        # Mulai countdown ketika waktu tinggal 5 detik
        if remaining_time <= 5 and not countdown_started:
            countdown_started = True
            countdown(5)
            dramatic_loss("Waktu habis! BOM meledak tepat setelah hitungan berakhir.")
            break

        remaining = high - low + 1
        if remaining <= 0:
            dramatic_loss("Kisaran angka tidak valid.")
            break

        try:
            guess = ask_number_in_range(low, high, guessed)
        except KeyboardInterrupt:
            console.print("\nPermainan dihentikan oleh pemain. Kembali ke menu.")
            time.sleep(1)
            break

        if guess == bomb:
            dramatic_loss(f"Anda menebak angka BOM ({bomb}). Permainan berakhir.")
            break

        guessed.append(guess)

        # Check win condition: only 1 number left and it's the bomb
        if remaining == 1 and low == bomb:
            secret_code = generate_secret_code(8)
            dramatic_win(secret_code)
            break

        # Check win condition: 2 numbers left and you guessed the non-bomb
        if remaining == 2:
            # If one of the remaining numbers is the bomb and you guessed the other one
            if (low == bomb and high == guess) or (high == bomb and low == guess):
                secret_code = generate_secret_code(8)
                dramatic_win(secret_code)
                break

        if guess > bomb:
            high = guess - 1
            console.print(f"Tebakan {guess} lebih besar dari angka BOM. Kisaran diperbarui menjadi [{low}..{high}].\n")
        else:
            low = guess + 1
            console.print(f"Tebakan {guess} lebih kecil dari angka BOM. Kisaran diperbarui menjadi [{low}..{high}].\n")

        remaining = high - low + 1
        if remaining == 2:
            console.print("Hati-hati — hanya tersisa 2 angka. Tebak salah satu yang bukan BOM untuk menang!")

    return secret_code

def secret_section(access_code_stored):
    clear()
    tampilkan_judul()
    console.print("[bold]Secret Section[/bold]\n")
    if not access_code_stored:
        console.print("Anda belum memiliki kode akses. Menangkan permainan untuk mendapatkan kode.\n")
        inquirer.text(message="Tekan Enter untuk kembali ke menu...").execute()
        return

    code = inquirer.text(message="Masukkan kode akses Secret Section:").execute()
    if code.strip() == access_code_stored:
        clear()
        tampilkan_judul()
        console.print("[bold green]Selamat! Anda berhasil memasuki Secret Section.[/bold green]\n")
        console.print("Ucapan: Terima kasih telah memainkan Angka BOM — Anda benar-benar ahli mengelola risiko!\n")
        console.print("Promosi: Kunjungi GitHub saya [underline]https://github.com/stn-marvel[/underline] untuk melihat proyek.")
        console.print("Jasa Development: Saya menyediakan pembuatan website (React, Tailwind, Firebase), integrasi, dan konsultasi. Hubungi saya jika tertarik.\n")
        inquirer.text(message="Tekan Enter untuk kembali ke menu...").execute()
    else:
        console.print(":no_entry_sign: Kode salah. Akses ditolak.\n")
        inquirer.text(message="Tekan Enter untuk kembali ke menu...").execute()

def main_menu():
    secret_code = None
    while True:
        clear()
        tampilkan_judul()
        choice = inquirer.select(
            message="Pilih opsi:",
            choices=["Permainan", "Peraturan", "Credits", "Secret Section", "Keluar"],
        ).execute()

        if choice == "Permainan":
            code = play_game()
            if code:
                secret_code = code
        elif choice == "Peraturan":
            show_rules()
        elif choice == "Credits":
            show_credits()
        elif choice == "Secret Section":
            secret_section(secret_code)
        elif choice == "Keluar":
            console.print("Terima kasih sudah bermain Angka BOM. Sampai jumpa!")
            break

if __name__ == "__main__":
    main_menu()
